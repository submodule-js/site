1 minute example

Install dependencies

npm install @submodule/core chalk

Write your color app

main.mjs
import { create } from "@submodule/core"
import chalk from "chalk"
 
const config = create(() => ({ color: process.env.LOG_COLOR || chalk.green }))
 
// craft an utilities with dependencies
export const logger = create((config) => {
  return (...args) => {
    console.log(...args.map(arg => config.color(arg)))
  }
}, config)
mymodule.mjs
import { logger } from "./main"
import { create } from "@submodule/core"
 
// access from everywhere
from(logger).execute(log => {
  log('hello', 'world')
})

Execute code

node main.mjs # should return hello world in green
LOG_COLOR=red node main.mjs # should return hello world in red
